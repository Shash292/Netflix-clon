pipeline {
    agent any
    stages {
        stage('Clone Repo') {
            steps {
                git branch: 'main', url: 'https://github.com/Shash292/Netflix-clon.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    dockerImage = docker.build("your-dockerhub-username/netflix-clone")
                }
            }
        }
        stage('Push to DockerHub') {
            steps {
                script {
                    docker.withRegistry('https://index.docker.io/v1/', 'dockerhub-credentials-id') {
                        dockerImage.push("latest")
                    }
                }
            }
        }
        stage('Deploy Container') {
            steps {
                sh 'docker rm -f netflix-clone || true'
                sh 'docker run -d -p 5000:5000 --name netflix-clone your-dockerhub-username/netflix-clone:latest'
            }
        }
    }
}
